persistent.save_data = {}  # 初始化为空字典  

# 保存状态  
def save_game_state(slot_id, character_attributes):  
    persistent.save_data[slot_id] = character_attributes  

# 加载状态  
def load_game_state(slot_id):  
    return persistent.save_data.get(slot_id, None)  # 如果没有找到存档，返回None
